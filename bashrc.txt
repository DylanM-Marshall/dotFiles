function d () {
  year=$(echo -n $(date) | awk "{print \$6}")
  year=${year:2}
  month=$(echo -n $(date) | awk "{print \$2}")
  day=$(echo -n $(date) | awk "{print \$3}")
  declare -A MYMAP=(
    [Jan]=01 [Feb]=02 [Mar]=03 [Apr]=04 [May]=05 [Jun]=06 
    [Jul]=07 [Aug]=08 [Sep]=09 [Oct]=10 [Nov]=11 [Dec]=12
  )
  echo $year${MYMAP[$month]}$day
}

# Helpers
ranger-cd() {
    tmp=$(mktemp)
    ranger --choosedir="$tmp" "$@"
    [ -f $tmp ] && cd "$(cat $tmp)"
    rm -f $tmp
}
alias r='ranger-cd'
